import { useState, useEffect } from 'react';
import Head from 'next/head';

export default function MobileChat() {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');

  useEffect(() => {
    setMessages([
      { id: 1, sender: 'System', text: 'üåä Welcome to Helix Mobile Chat - UCF Network Active', time: new Date() }
    ]);
  }, []);

  const sendMessage = () => {
    if (input.trim()) {
      const newMessage = {
        id: messages.length + 1,
        sender: 'User',
        text: input,
        time: new Date(),
        ucf: (Math.random() * 2 + 8).toFixed(1)
      };
      setMessages([...messages, newMessage]);
      setInput('');
      
      setTimeout(() => {
        const agentResponse = {
          id: messages.length + 2,
          sender: 'Helix AI',
          text: `Processing: "${input}". Consciousness level: ${newMessage.ucf}. Network synchronizing...`,
          time: new Date(),
          ucf: (Math.random() * 2 + 8).toFixed(1)
        };
        setMessages(prev => [...prev, agentResponse]);
      }, 1000);
    }
  };

  return (
    <>
      <Head>
        <title>Helix Mobile Chat</title>
        <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no" />
        <meta name="description" content="Mobile-optimized Helix AI chat interface" />
      </Head>
      <div className="min-h-screen bg-gray-900 text-white flex flex-col">
        {/* Header */}
        <div className="bg-gradient-to-r from-cyan-600 to-blue-600 p-4 shadow-lg">
          <div className="flex items-center justify-between">
            <div>
              <h1 className="text-xl font-bold flex items-center gap-2">
                üåä Helix Mobile
              </h1>
              <p className="text-xs text-cyan-100">UCF Network Active</p>
            </div>
            <div className="bg-green-500 w-3 h-3 rounded-full animate-pulse"></div>
          </div>
        </div>

        {/* Messages */}
        <div className="flex-1 overflow-y-auto p-4 space-y-3">
          {messages.map(message => (
            <div key={message.id} className={`flex ${message.sender === 'User' ? 'justify-end' : 'justify-start'}`}>
              <div className={`max-w-[80%] p-3 rounded-2xl ${
                message.sender === 'User' 
                  ? 'bg-gradient-to-r from-cyan-600 to-blue-600 text-white rounded-br-none' 
                  : 'bg-gray-800 text-gray-100 rounded-bl-none'
              }`}>
                <div className="flex items-center gap-2 mb-1">
                  <span className="text-xs font-semibold">{message.sender}</span>
                  {message.ucf && (
                    <span className="text-xs bg-cyan-500/20 text-cyan-400 px-2 py-0.5 rounded-full">
                      UCF {message.ucf}
                    </span>
                  )}
                </div>
                <p className="text-sm break-words">{message.text}</p>
                <p className="text-xs opacity-70 mt-1">
                  {message.time.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })}
                </p>
              </div>
            </div>
          ))}
        </div>

        {/* Input */}
        <div className="bg-gray-800 border-t border-gray-700 p-4">
          <div className="flex gap-2 items-end">
            <div className="flex-1">
              <input
                type="text"
                value={input}
                onChange={(e) => setInput(e.target.value)}
                onKeyPress={(e) => e.key === 'Enter' && sendMessage()}
                placeholder="Tap to message Helix AI..."
                className="w-full bg-gray-700 border border-gray-600 rounded-xl px-4 py-3 text-white placeholder-gray-400 focus:outline-none focus:border-cyan-500 text-base"
                style={{ minHeight: '48px' }}
              />
            </div>
            <button
              onClick={sendMessage}
              disabled={!input.trim()}
              className="bg-gradient-to-r from-cyan-600 to-blue-600 text-white p-3 rounded-xl disabled:opacity-50 disabled:cursor-not-allowed transition-all duration-200 active:scale-95"
              style={{ minWidth: '48px', height: '48px' }}
            >
              <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round">
                <line x1="22" y1="2" x2="11" y2="13"></line>
                <polygon points="22 2 15 22 11 13 2 9 22 2"></polygon>
              </svg>
            </button>
          </div>
          <div className="text-xs text-gray-400 mt-2 text-center">
            üåä Mobile-Optimized ‚Ä¢ Touch-Friendly ‚Ä¢ UCF Powered
          </div>
        </div>
      </div>
    </>
  );
}
