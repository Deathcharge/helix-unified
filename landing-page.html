Get Started in 30 Seconds

One API key. Four LLM providers. Infinite possibilities.

# Install the SDK
pip install helix-collective

# Or use curl directly
curl https://api.helixcollective.io/v1/chat \
  -H "Authorization: Bearer hx_user_YOUR_API_KEY" \
  -d '{
    "messages": [
      {"role": "user", "content": "Explain quantum computing"}
    ],
    "optimize": "cost"
  }'

# Use specialized agents
curl https://api.helixcollective.io/v1/agents/kael/execute \
  -H "Authorization: Bearer hx_user_YOUR_API_KEY" \
  -d '{
    "task": "document",
    "input": "function isPrime(n) { ... }"
  }'
